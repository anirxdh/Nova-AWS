```typescript
export interface DomSnapshot {
  url: string;
  title: string;
  buttons: ElementInfo[];
  links: ElementInfo[];
  inputs: InputInfo[];
  forms: FormInfo[];
  text_content: string;
}

interface ElementInfo {
  selector: string;
  text: string;
  visible: boolean;
}

interface InputInfo {
  selector: string;
  type: string;
  value: string;
  placeholder: string;
  visible: boolean;
}

interface FormInfo {
  selector: string;
  action: string;
  method: string;
  inputs: InputInfo[];
}
```

The `scrapeDom()` function should:
- Set `url` to `window.location.href` and `title` to `document.title`
- Query all `button` and `[role="button"]` elements. For each, generate a CSS selector using `buildSelector()` (see below), get `textContent.trim()`, and check visibility with `isVisible()`. Limit to first 50.
- Query all `a[href]` elements. Same treatment. Limit to first 50.
- Query all `input, textarea, select` elements. Capture type, value, placeholder. Limit to first 30.
- Query all `form` elements. Capture action, method, and nested inputs. Limit to first 10.
- Set `text_content` to `document.body?.innerText?.slice(0, 3000) || ''` (truncated to keep payload manageable)

Helper functions:
- `isVisible(el: Element): boolean` -- checks `el.checkVisibility?.() ?? true` (checkVisibility is widely supported in Chrome), AND `el.getBoundingClientRect()` has width > 0 and height > 0.
- `buildSelector(el: Element): string` -- priority order: (1) if el has an id, return `#${CSS.escape(el.id)}`; (2) if el has unique data-testid or aria-label, use `[data-testid="..."]` or `[aria-label="..."]`; (3) construct `tagName.className` path with nth-child if needed. Keep it simple and robust -- the selector must work with `document.querySelector()`. Maximum 3 levels of ancestor traversal.

**1. Add `sendTask()` to `src/background/api/backend-client.ts`**:

```typescript
export interface TaskResponse {
  type: 'answer' | 'steps';
  text?: string;
  actions?: Array<{
    action: string;
    selector?: string;
    value?: string;
    url?: string;
    direction?: string;
    description: string;
  }>;
}

/**
 * Send a task (command + screenshot + DOM) to the backend for Nova 2 Lite reasoning.
 */
export async function sendTask(
  command: string,
  screenshotDataUrl: string,
  domSnapshot: object
): Promise<TaskResponse> {
  // Strip the data:image/png;base64, prefix to get raw base64
  const base64 = screenshotDataUrl.replace(/^data:image\/\w+;base64,/, '');

  const response = await fetch(`${BACKEND_URL}/task`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      command,
      screenshot: base64,
      dom_snapshot: domSnapshot,
    }),
  });

  if (!response.ok) {
    let detail = '';
    try {
      const errBody = await response.json();
      detail = errBody?.detail || JSON.stringify(errBody);
    } catch {
      detail = `HTTP ${response.status}`;
    }

    if (response.status === 500 && detail.includes('credentials')) {
      throw new Error('Backend AWS credentials not configured — check backend/.env');
    }
    throw new Error(`Task processing failed — ${detail}`);
  }

  return await response.json();
}
```
